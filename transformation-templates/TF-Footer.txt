<!-- Added 12.2013 by R. Taylor to sanitize search field input against whitelist (per CERT) on keypress and on blur. Allows both CR/LF and BACK/Del -->
<script>
 
    $( document ).ready(function() {

        $('#keywords').keypress(function (e) {
    		var regex = new RegExp("^[a-zA-Z0-9'\-.*+ \s\b]{0,1}$");
    		var str = String.fromCharCode(!e.charCode ? e.which : e.charCode);
    		if (regex.test(str) || e.which == 13 || e.which == 8) {
        	     return true;
   		 }

    		e.preventDefault();
    		return false;
	});

        $('#keywords').bind('input', function(e) {
           if ( !(/^[a-zA-Z0-9'\-.*+ \s\b]+{0,1}$/.test($(this).val())) )
           {
              $('#keywords').val('Use alphanumeric only');
           }
	}); 

        $('#keywords1').keypress(function (e) {

        	     return true;
   		}

    		e.preventDefault();
    		return false;
	});

        $('#keywords1').bind('input', function(e) {
           if ( !(/^[a-zA-Z0-9'\-.*+ \s\b]+{0,1}$/.test($(this).val())) )

           }
	}); 

    });
